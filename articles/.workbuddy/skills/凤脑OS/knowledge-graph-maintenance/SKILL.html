维护痛点解决

质量评分计算

质量评分算法（伪代码）

def calculate_quality_score(node, links):
    # 完整性评分（30%）
    completeness = check_completeness(node)
    completeness_score = completeness * 0.3
    
    # 准确性评分（25%）
    accuracy = verify_accuracy(node)
    accuracy_score = accuracy * 0.25
    
    # 关联有效性评分（20%）
    validity = validate_links(node, links)
    validity_score = validity * 0.2
    
    # 清晰度评分（15%）
    clarity = assess_clarity(node)
    clarity_score = clarity * 0.15
    
    # 时效性评分（10%）
    timeliness = check_timeliness(node)
    timeliness_score = timeliness * 0.1
    
    # 总分（1-10分）
    total_score = (
        completeness_score + 
        accuracy_score + 
        validity_score + 
        clarity_score + 
        timeliness_score
    ) * 10
    
    return round(total_score, 2)
