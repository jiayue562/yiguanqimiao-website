## 批量创建脚本

```powershell
# 创建 100 个思维模型目录结构
$models = @(
    "001-第一性原理", "002-MECE 分析法", "003-金字塔原理", "004-逻辑树", "005-因果思维",
    "006-系统思维", "007-二阶思维", "008-概率思维", "009-逆向思维", "010-成长思维",
    # ... 继续添加 90 个模型
)

foreach ($model in $models) {
    $basePath = "C:\Users\jia'yue\.agents\skills\龙脑 OS\思维模型\$model"
    New-Item -ItemType Directory -Path "$basePath\references" -Force
    New-Item -ItemType Directory -Path "$basePath\triggers" -Force
    
    # 创建简版 SKILL.md
    $template = Get-Content "template.md" -Raw
    $template = $template -replace "\[模型名称\]", $model
    $template | Out-File -FilePath "$basePath\SKILL.md" -Encoding UTF8
}
```
